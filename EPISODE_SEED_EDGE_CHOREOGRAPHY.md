### **Choreography System**

```typescript
interface EpisodeChoreography {
  seed: Episode;
  edge: Episode;
  nextSeed?: Episode; // Cycles back or continues
  nested?: EpisodeChoreography[]; // Recursive nesting
  cycle?: Episode; // Cycle target
}

// Example:
const episode5: EpisodeChoreography = {
  seed: Episode5,
  edge: Episode6,
  nextSeed: Episode6, // Episode 6 becomes seed for Episode 7
  nested: [
    {
      seed: Episode4, // Previous episode
      edge: Episode5,
    },
    {
      seed: Episode5,
      edge: Episode6,
    },
    {
      seed: Episode6,
      edge: Episode7, // Next episode
    }
  ],
  cycle: Episode1 // Can cycle back to Episode 1
};
```

### **Recursive Nesting**

```typescript
function nestRecursively(episode: Episode, level: number): EpisodeChoreography {
  if (level === 0) {
    return {
      seed: episode,
      edge: episode.next,
    };
  }
  
  return {
    seed: episode,
    edge: episode.next,
    nested: [
      nestRecursively(episode, level - 1), // Recursive nesting
      nestRecursively(episode.next, level - 1),
    ],
  };
}
```
